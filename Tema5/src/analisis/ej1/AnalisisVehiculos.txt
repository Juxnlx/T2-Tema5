abstract Vehiculos
	- marca: String
	- modelo: String
	- color: String
	- matricula: String
	- estadoMotor: boolean 
	- marchaActual: int 
	- velocidadActual: double
	---------------------------------
	+ Vehiculos(marca: String, modelo: String, color: String, matricula: String)
	+ getMarchaActual(): int
	+ getVelocidadActual(): double
	+ setVelocidad(double velocidad): void
	+ parar(): void
	+ arrarcar(): void
	+ subirMarcha(): void
	+ bajarMarcha(): void 
	+ ajustarVelocidad(): void
